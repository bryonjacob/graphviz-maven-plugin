18  /**
19   * provides common functionality for the various mojos that ultimately use graphviz to render
20   * images.
21   */
22  public abstract class AbstractDotMojo extends AbstractMojo {
23      /**
24       * pattern of files to exclude (separate multiple patterns with commas).
25       * @parameter
26       */
27      protected String excludes;
28      /**
29       * layout to use.
30       *
31       * this can be any value that can be provided to the -K option of the graphviz binary - at
32       * time of writing this is <code>dot|neato|twopi|circo|fdp|sfdp</code>. see the man pages
33       * (http://www.graphviz.org/pdf/dot.1.pdf) for descriptions of the formats.
34       * @parameter default-value="dot"
35       * @required
36       */
37      protected String layout;
38      /**
39       * output format.
40       *
41       * this can be any value that can be provided to the -T option of the graphviz binary - the
42       * list of all known options is here:
43       * <a href="http://www.graphviz.org/doc/info/output.html">http://www.graphviz.org/doc/info/output.html</a>.
44       * the set of options available on your maching may vary - run <code>dot -T?</code> for a list
45       * of output formats supported by your installation.
46       * @parameter default-value="png"
47       * @required
48       */
49      protected String output;
50      /**
51       * destination directory into which rendered output will be written.
52       * @parameter expression="${project.build.directory}/graphviz"
53       * @required
54       */
55      protected File destdir;
56      /**
57       * the graphviz binary to use.
58       *
59       * by default, it is assumed that "dot" is on your PATH - if not, this value should be set to
60       * the full path to the dot binary.
61       * @parameter default-value="dot"
62       * @required
63       */
64      protected String dot;
65      /**
66       * file mapper to use.
67       *
68       * provide a custom mapper used to transform input filenames into output - by default, the
69       * mapper used will be a FileExtensionMapper, using the value of <code>output</code> as the
70       * extension on the output file.
71       * @parameter
72       */
73      protected FileMapper mapper;
74  
75      // the basedir and includes vary across the concrete implementations, so we need to access
76      // these values through getters.
77      protected abstract File getBasedir();
78      protected abstract String getIncludes();
79  
80      // by default, we do an identity transform and just pass the original input file to graphviz -
81      // subclasses can override this where the original file must be somehow transformed before
82      // passing it to graphviz.
83      protected File transformInputFile(File input) throws MojoExecutionException { return input; }
84  
85      public void execute() throws MojoExecutionException, MojoFailureException {
86          try {
87              // find matching files
88              List<String> filenames = FileUtils.getFileNames(
89                      getBasedir(), getIncludes(), excludes, false);
90              // get the configured mapper, or create a default one if none is specified
91              FileMapper mapper = this.mapper == null ? createDefaultMapper() : this.mapper;
92              // for each file...
93              for (String filename : filenames) {
94                  // compute the "from" and "to" files
95                  File from = new File(getBasedir(), filename);
96                  File to = new File(destdir, mapper.getMappedFileName(filename));
97                  getLog().debug(String.format("converting %s to %s", from, to));
98                  // make sure the directory for the "to" file exists
99                  to.getParentFile().mkdirs();
100                 // call graphviz to convert the "from" file
101                 Commandline cli = new Commandline();
102                 cli.setExecutable(dot);
103                 cli.addArguments(new String[] {
104                         String.format("-K%s", layout),
105                         String.format("-T%s", output),
106                         transformInputFile(from).getAbsolutePath()
107                 });
108                 try {
109                     // pipe the output stream from graphviz to the "to" file
110                     Process process = cli.execute();
111                     IOUtil.copy(process.getInputStream(), new FileOutputStream(to));
112                 } catch (CommandLineException e) {
113                     throw new MojoExecutionException("error performing conversion", e);
114                 }
115             }
116 
117         } catch (IOException e) {
118             throw new MojoExecutionException(
119                     String.format("error finding files from %s (includes = %s, excludes = %s)",
120                             getBasedir(), getIncludes(), excludes), e);
121         }
122     }
123 
124     // by default, map files by using the "output" format as the extension
125     protected FileExtensionMapper createDefaultMapper() {
126         FileExtensionMapper fileExtensionMapper = new FileExtensionMapper();
127         fileExtensionMapper.setTargetExtension(output);
128         return fileExtensionMapper;
129     }
