14  /**
15   * use DOTML and GraphViz to render DOTML .xml files to images.
16   *
17   * This mojo does much the same as the <a href="/dot-mojo.html">graphviz:dot</a> mojo, except that
18   * instead of standard Graphviz .dot files, it expects its input to be DOTML files, an XML dialect
19   * for .dot files.
20   *
21   * This product includes DOTML developed by Martin Loetzsch
22   * (<a href="http://www.martin-loetzsch.de/DOTML">http://www.martin-loetzsch.de/DOTML</a>).
23   *
24   * @goal dotml
25   * @phase generate-sources
26   */
27  public class DotmlMojo extends AbstractDotMojo {

29      /**
30       * the base directory in which .dot files can be found.

33       * @required
34       */
35      protected File basedir;
36      /**
37       * pattern of files to include (separate multiple patterns with commas).
38       *
39       * @parameter expression="**\/*.xml"
40       * @required
41       */
42      protected String includes;
43  
44      protected File getBasedir() {
45          return basedir;
46      }
47  
48      protected String getIncludes() {
49          return includes;
50      }
51  
52      private final TransformerFactory transformerFactory = TransformerFactory.newInstance();
53  
54      // when we are using DOTML files, we need to transform them to DOT files first.
55      protected File transformInputFile(File from) throws MojoExecutionException {
56          // create a temp file
57          File tempFile;
58          try {
59              tempFile = File.createTempFile("dotml-tmp", ".xml");
60          } catch (IOException e) {
61              throw new MojoExecutionException(
62                      "error creating temp file to hold DOTML to DOT translation", e);
63          }
64          // perform an XSLT transform from the input file to the temp file
65          Source xml = new StreamSource(from);
66          Result result = new StreamResult(tempFile);
67          try {
68              Source xslt = new StreamSource(
69                      getClass().getClassLoader().getResourceAsStream("dotml/dotml2dot.xsl"));
70              transformerFactory.newTransformer(xslt).transform(xml, result);
71          } catch (TransformerException e) {
72              throw new MojoExecutionException(
73                      String.format("error transforming %s from DOTML to DOT file", from), e);
74          }
75          // return the temp file
76          return tempFile;
77      }
